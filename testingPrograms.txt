void Insert(int A, int n){
    int temp = A[n];

    while(n > 0 && temp > A[n/2]){
        A[n] = A[n/2];
        n = n / 2;
    }

    A[n] = temp;
}

void Delete(int A[], int &n){
    int val, x, i, j, temp;

    val = A[0];
    x = A[--n];
    A[0] = x;

    i = 0;
    j = 2 * i + 1;

    while(j < n){
        if((j + 1) < n && A[j + 1] > A[j]){
            j = j + 1;
        }

        if(A[i] < A[j]){
            temp = A[i];
            A[i] = A[j];
            A[j] = temp;

            i = j;
            j = 2 * i + 1;
        }
        else{
            break;
        }
    }
    return val;
}
